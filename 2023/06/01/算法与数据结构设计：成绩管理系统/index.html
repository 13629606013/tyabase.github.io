# 实验周习题记录：成绩管理系统

💁问题描述：给出 n 个学生的考试成绩表，成绩表包括学生的学号、姓名、考试成绩（高等数学、英语、物理），设计一个简单的成绩管理程序。

基本要求：

建立成绩表，能够插入、删除、修改学生的成绩记录；

🌟按任一单科成绩排序；

🌟计算每名学生的平均成绩；

🌟统计任一单科成绩不及格的学生人数，输出不及格人数及不及格的学生名单

🌟根据平均成绩将成绩表按由高到低的次序排列，统计每名学生在考试中获得的名次，分数相同的为同一名次，按名次输出成绩表。

🌟成绩表保存在文件中，可以从文件读取数据。

测试数据：学生可以根据自己班级的考试成绩单，任意截取一部分做为测试数据。

#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>
using namespace std;

// 学生结构体
struct student {
    string id; // 学号
    string name; // 姓名
    double math; // 高等数学成绩
    double eng; // 英语成绩
    double phy; // 物理成绩
    double avg; // 平均成绩
    int rank; // 排名
};

vector<student> students; // 存储学生信息的vector

// 按学号查找学生在vector中的索引
int find_idx(string id) {
    for (int i = 0; i < students.size(); i++) {
        if (students[i].id == id) {
            return i;
        }
    }
    return -1; // 没找到
}

// 添加学生
void add_student() {
    student s;
    cout << "请输入学生信息：" << endl;
    cout << "学生学号：";
    cin >> s.id;
    cout << "学生姓名：";
    cin >> s.name;
    cout << "高数成绩：";
    cin >> s.math;
    cout << "英语成绩：";
    cin >> s.eng;
    cout << "物理成绩：";
    cin >> s.phy;
    s.avg = (s.math + s.eng + s.phy) / 3;
    students.push_back(s);
}

// 删除学生
void delete_student() {
    string id;
    cout << "请输入要删除的学生的学号：";
    cin >> id;
    int idx = find_idx(id);
    if (idx != -1) {
        students.erase(students.begin() + idx);
        cout << "删除成功！" << endl;
    }
    else {
        cout << "找不到该学生！" << endl;
    }
}

// 修改学生成绩
void modify_student() {
    string id;
    cout << "请输入要修改的学生的学号：";
    cin >> id;
    int idx = find_idx(id);
    if (idx != -1) {
        student& s = students[idx];
        cout << "请输入新的成绩：" << endl;
        cout << "高数成绩：";
        cin >> s.math;
        cout << "英语成绩：";
        cin >> s.eng;
        cout << "物理成绩：";
        cin >> s.phy;
        s.avg = (s.math + s.eng + s.phy) / 3;
        cout << "修改成功！" << endl;
    }
    else {
        cout << "找不到该学生！" << endl;
    }
}

// 按任一单科成绩排序
bool cmp_math(const student& a, const student& b) {
    return a.math > b.math;
}
bool cmp_eng(const student& a, const student& b) {
    return a.eng > b.eng;
}
bool cmp_phy(const student& a, const student& b) {
    return a.phy > b.phy;
}
void sort_by_score() {
    int choice;
    cout << "请选择排序方式（1. 高数；2. 英语；3. 物理）：";
    cin >> choice;
    switch (choice) {
    case 1:
        sort(students.begin(), students.end(), cmp_math);
        cout << "按高数成绩排序后的成绩表：" << endl;
        break;
    case 2:
        sort(students.begin(), students.end(), cmp_eng);
        cout << "按英语成绩排序后的成绩表：" << endl;
        break;
    case 3:
        sort(students.begin(), students.end(), cmp_phy);
        cout << "按物理成绩排序后的成绩表：" << endl;
        break;
    default:
        cout << "输入错误！" << endl;
        return;
    }

    for (int i = 0; i < students.size(); i++) {
        cout << students[i].id << "\t";
        cout << students[i].name << "\t";
        cout << students[i].math << "\t";
        cout << students[i].eng << "\t";
        cout << students[i].phy << "\t";
        cout << students[i].avg << endl;
    }
}

// 统计某一科不及格人数并输出不及格学生名单
void count_failures() {
    int choice;
    int cnt = 0;
    vector<string> failure_list; // 不及格学生名单
    cout << "请选择科目（1. 高等数学；2. 英语；3. 物理）：";
    cin >> choice;
    switch (choice) {
    case 1:
        for (int i = 0; i < students.size(); i++) {
            if (students[i].math < 60) {//判断合格or不合格
                cnt++;
                failure_list.push_back(students[i].name);
            }
        }
        cout << "高数不及格人数：" << cnt << endl;
        cout << "不及格学生名单：" << endl;
        break;
    case 2:
        for (int i = 0; i < students.size(); i++) {
            if (students[i].eng < 60) {
                cnt++;
                failure_list.push_back(students[i].name);
            }
        }
        cout << "英语不及格人数：" << cnt << endl;
        cout << "不及格学生名单：" << endl;
        break;
    case 3:
        for (int i = 0; i < students.size(); i++) {
            if (students[i].phy < 60) {
                cnt++;
                failure_list.push_back(students[i].name);
            }
        }
        cout << "物理不及格人数：" << cnt << endl;
        cout << "不及格学生名单：" << endl;
        break;
    default:
        cout << "输入错误！" << endl;
        return;
    }

    for (int i = 0; i < failure_list.size(); i++) {
        cout << failure_list[i] << endl;
    }
}

// 按平均成绩排序并输出名次
bool cmp_avg(const student& a, const student& b) {
    return a.avg > b.avg;
}
void sort_by_avg() {
    sort(students.begin(), students.end(), cmp_avg);
    cout << "按平均成绩排序后的成绩表：" << endl;
    cout << "排名\t学号\t姓名\t高等数学\t英语\t物理\t平均成绩" << endl;
    int rank = 1;
    for (int i = 0; i < students.size(); i++) {
        if (i > 0 && students[i].avg != students[i - 1].avg) {
            rank = i + 1;
        }
        students[i].rank = rank;
        cout << students[i].rank << "\t";
        cout << students[i].id << "\t";
        cout << students[i].name << "\t";
        cout << students[i].math << "\t";
        cout << students[i].eng << "\t";
        cout << students[i].phy << "\t";
        cout << students[i].avg << endl;
    }
}

// 从文件读取数据
void read_data() {
    ifstream fin("scores.txt");
    if (!fin) {
        cout << "无法打开文件！" << endl;
        return;
    }
    student s;
    while (fin >> s.id >> s.name >> s.math >> s.eng >> s.phy) {
        s.avg = (s.math + s.eng + s.phy) / 3;
        students.push_back(s);
    }
    fin.close();
    cout << "从文件读取数据成功！" << endl;
}

// 将数据保存到文件
void save_data() {
    ofstream fout("scores.txt");
    if (!fout) {
        cout << "无法创建文件！" << endl;
        return;
    }
    for (int i = 0; i < students.size(); i++) {
        fout << students[i].id << " ";
        fout << students[i].name << " ";
        fout << students[i].math << " ";
        fout << students[i].eng << " ";
        fout << students[i].phy << endl;
    }
    fout.close();
    cout << "数据保存到文件成功！" << endl;
}

int main() {
    int choice;
    while (true) {
        cout << endl;
        cout << "成绩管理程序" << endl;
        cout << "#1  添加学生" << endl;
        cout << "#2  删除学生" << endl;
        cout << "#3  修改学生成绩" << endl;
        cout << "#4  按任一单科成绩排序" << endl;
        cout << "#5  计算每名学生的平均成绩" << endl;
        cout << "#6  统计任一单科成绩不及格的学生人数" << endl;
        cout << "#7  根据平均成绩将成绩表按由高到低的次序排列，统计每名学生在考试中获得的名次" << endl;
        cout << "#8  从文件读取数据" << endl;
        cout << "#9  将数据保存到文件" << endl;
        cout << "#10 退出程序" << endl;
        cout << "请选择序号操作：";
        cin >> choice;
        switch (choice) {
        case 1:
            add_student();
            break;
        case 2:
            delete_student();
            break;
        case 3:
            modify_student();
            break;
        case 4:
            sort_by_score();
            break;
        case 5:
            for (int i = 0; i < students.size(); i++) {
                students[i].avg = (students[i].math + students[i].eng + students[i].phy) / 3;
                cout << students[i].name << " 的平均成绩为：" << students[i].avg << endl;
            }
            break;
        case 6:
            count_failures();
            break;
        case 7:
            sort_by_avg();
            break;
        case 8:
            read_data();
            break;
        case 9:
            save_data();
            break;
        case 10:
            cout << "程序已退出。" << endl;
            return 0;
        default:
            cout << "输入错误！" << endl;
            break;
        }
    }
}
